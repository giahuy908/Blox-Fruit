-- Hệ thống Dịch Chuyển mượt mà (Tween) bằng CFrame
local TweenSpeed = 325
local CurrentTween = nil

local function stopTeleport()
    if CurrentTween then
        CurrentTween:Cancel()
        CurrentTween = nil
    end
    local char = Plr.Character
    if char and char:FindFirstChild("PartTele") then
        char.PartTele:Destroy()
    end
end

local function topos(targetCFrame)
    local char = Plr.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") or char.Humanoid.Health <= 0 then return end
    
    local hrp = char.HumanoidRootPart
    local distance = (targetCFrame.Position - hrp.Position).Magnitude
    
    if distance < 10 then
        hrp.CFrame = targetCFrame
        return
    end

    -- Tạo một Part đệm để nhân vật đứng khi dịch chuyển xuyên tường
    if not char:FindFirstChild("PartTele") then
        local part = Instance.new("Part", char)
        part.Size = Vector3.new(10, 1, 10)
        part.Name = "PartTele"
        part.Anchored = true
        part.Transparency = 1
        part.CanCollide = true
        part.CFrame = hrp.CFrame
    end

    local partTele = char.PartTele
    partTele.CFrame = hrp.CFrame

    -- Liên kết sự kiện di chuyển nhân vật cùng Part đệm
    local connection
    connection = partTele:GetPropertyChangedSignal("CFrame"):Connect(function()
        if char:FindFirstChild("HumanoidRootPart") then
            hrp.CFrame = partTele.CFrame
        else
            connection:Disconnect()
        end
    end)

    if CurrentTween then CurrentTween:Cancel() end
    
    local tweenInfo = TweenInfo.new(distance / TweenSpeed, Enum.EasingStyle.Linear)
    CurrentTween = TweenService:Create(partTele, tweenInfo, {CFrame = targetCFrame})
    CurrentTween:Play()

    CurrentTween.Completed:Connect(function(state)
        if state == Enum.PlaybackState.Completed then
            if char:FindFirstChild("PartTele") then
                char.PartTele:Destroy()
            end
            connection:Disconnect()
        end
    end)
end

local function TP1(targetCFrame)
    topos(targetCFrame)
end

-- Trang bị vũ khí tự động
local function EquipWeapon(weaponType)
    local char = Plr.Character
    local backpack = Plr.Backpack
    if not char or not char:FindFirstChild("Humanoid") then return end

    -- Tìm vũ khí theo loại trong Backpack
    for _, tool in pairs(backpack:GetChildren()) do
        if tool:IsA("Tool") then
            if (weaponType == "Melee" and tool.ToolTip == "Melee") or
               (weaponType == "Sword" and tool.ToolTip == "Sword") or
               (weaponType == "Gun" and tool.ToolTip == "Gun") or
               (weaponType == "Blox Fruit" and tool.ToolTip == "Blox Fruit") or
               (tool.Name == weaponType) then
                char.Humanoid:EquipTool(tool)
                break
            end
        end
    end
end

local function UnEquipWeapon()
    local char = Plr.Character
    if not char or not char:FindFirstChild("Humanoid") then return end
    char.Humanoid:UnequipTools()
end

-- Thuật toán lấy Đảo hiện tại trong Raid
local function GetRaidIsland(num)
    local closest, dist = nil, math.huge
    for _, island in pairs(workspace._WorldOrigin.Locations:GetChildren()) do
        if island.Name == "Island " .. num then
            local mag = (island.Position - HRP().Position).Magnitude
            if mag < dist then
                dist = mag
                closest = island
            end
        end
    end
    return closest
end

local function GetNextRaidIsland()
    for _, num in ipairs({5, 4, 3, 2, 1}) do
        local island = GetRaidIsland(num)
        if island and (island.Position - HRP().Position).Magnitude <= 4000 then
            return island
        end
    end
end

local ESPPlayer = false
local ESPConnections = {}

-- Chức năng ESP người chơi cao cấp
local function RemoveESP(player)
    if player.Character then
        local hrp = player.Character:FindFirstChild("HumanoidRootPart")
        if hrp and hrp:FindFirstChild("PlayerESP") then
            hrp.PlayerESP:Destroy()
        end
    end
    if ESPConnections[player] then
        ESPConnections[player]:Disconnect()
        ESPConnections[player] = nil
    end
end

local function CreateESP(player)
    if player == Plr then return end
    if not player.Character then return end

    local char = player.Character
    local hrp = char:WaitForChild("HumanoidRootPart", 3)
    local hum = char:WaitForChild("Humanoid", 3)
    if not hrp or not hum then return end

    RemoveESP(player)

    local gui = Instance.new("BillboardGui", hrp)
    gui.Name = "PlayerESP"
    gui.Size = UDim2.new(0, 200, 0, 50)
    gui.StudsOffset = Vector3.new(0, 3, 0)
    gui.AlwaysOnTop = true

    local nameLabel = Instance.new("TextLabel", gui)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Size = UDim2.new(1, 0, 0.5, 0)
    nameLabel.TextSize = 14
    nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    nameLabel.TextStrokeTransparency = 0
    nameLabel.Font = Enum.Font.SourceSansBold

    local hpLabel = Instance.new("TextLabel", gui)
    hpLabel.BackgroundTransparency = 1
    hpLabel.Size = UDim2.new(1, 0, 0.5, 0)
    hpLabel.Position = UDim2.new(0, 0, 0.5, 0)
    hpLabel.TextSize = 14
    hpLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
    hpLabel.TextStrokeTransparency = 0
    hpLabel.Font = Enum.Font.SourceSansBold

    ESPConnections[player] = RunService.RenderStepped:Connect(function()
        if not ESPPlayer or not char or not hrp or hum.Health <= 0 then
            RemoveESP(player)
            return
        end
        local myHrp = HRP()
        if myHrp then
            local dist = math.floor((myHrp.Position - hrp.Position).Magnitude)
            nameLabel.Text = player.Name .. " [" .. dist .. "m]"
            hpLabel.Text = "[" .. math.floor(hum.Health) .. "/" .. math.floor(hum.MaxHealth) .. "]"
        end
    end)
end

ESPSection:CreateToggle("ESP Người chơi", false, function(state)
    ESPPlayer = state
    if state then
        for _, p in pairs(Players:GetPlayers()) do
            CreateESP(p)
        end
        Players.PlayerAdded:Connect(CreateESP)
        Players.PlayerRemoving:Connect(RemoveESP)
    else
        for _, p in pairs(Players:GetPlayers()) do
            RemoveESP(p)
        end
    end
end)

local CharacterSection = MiscTab:CreateSection("Năng lực nhân vật")

CharacterSection:CreateSlider("Tốc độ chạy (WalkSpeed)", 16, 250, 16, function(value)
    _G.WalkSpeedValue = value
    local hum = Plr.Character and Plr.Character:FindFirstChild("Humanoid")
    if hum then hum.WalkSpeed = value end
end)

CharacterSection:CreateSlider("Sức nhảy (JumpPower)", 50, 400, 50, function(value)
    _G.JumpValue = value
    local hum = Plr.Character and Plr.Character:FindFirstChild("Humanoid")
    if hum then hum.JumpPower = value end
end)

CharacterSection:CreateToggle("Đi trên nước", false, function(state)
    _G.WalkOnWater = state
end)

task.spawn(function()
    while task.wait(0.2) do
        pcall(function()
            if _G.WalkOnWater then
                -- Đơn giản hóa cơ chế noclip trên nước
                local platform = workspace:FindFirstChild("WaterPlatform")
                if not platform then
                    platform = Instance.new("Part", workspace)
                    platform.Name = "WaterPlatform"
                    platform.Size = Vector3.new(9999, 1, 9999)
                    platform.Anchored = true
                    platform.Transparency = 1
                end
                platform.CFrame = CFrame.new(HRP().Position.X, 0, HRP().Position.Z)
            else
                if workspace:FindFirstChild("WaterPlatform") then
                    workspace.WaterPlatform:Destroy()
                end
            end
        end)
    end
end)

            v.Material = Enum.Material.SmoothPlastic
